2.dd
-----------------------------------------------
dd if=/dev/zero of=junk.data bs=1M count=1
dd if=/dev/cdrom of=image.iso
mkisofs -V "Label" -o image.iso source_dir/
isohybrid image.iso
dd if=image.iso of=/dev/sdb1  OR cat image.iso > /dev/sdb1
cdrecord -v /dev/cdrom image.iso
-----------------------------------------------
